31  /**
32   * <p>Provides a simple interface for interacting with the public Guild
33   * Wars 2 API. Whenever possible, methods of this interface will return simple
34   * objects. However, if there is a possibility for an error being returned by
35   * the remote API, then the resulting object of this client's method will be
36   * wrapped in an {@link com.google.common.base.Optional}. In all such cases,
37   * an instance of {@link com.jrfom.gw2.api.model.GwApiError} will be thrown.
38   * This error can either be ignored, or it can be trapped with a
39   * {@code try...catch} construct.</p>
40   *
41   * <p>You can either instantiate this class directly, or use the
42   * {@link ApiClient#getInstance()} method to retrieve a static instance.</p>
43   */
44  @Component
45  public class ApiClient extends RestTemplate {
46    private static final Logger log = LoggerFactory.getLogger(ApiClient.class);
47    private final String baseUrl = "https://api.guildwars2.com/v1/";
48  
49    private static ApiClient apiClient;
50  
51    public ApiClient() {
52      ResponseErrorHandler errorHandler = new ResponseErrorHandler() {
53        @Override
54        public boolean hasError(ClientHttpResponse response) throws IOException {
55          HttpStatus status = response.getStatusCode();
56          HttpStatus.Series series = status.series();
57          return (HttpStatus.Series.CLIENT_ERROR.equals(series) ||
58            HttpStatus.Series.SERVER_ERROR.equals(series)
59          );
60        }
61  
62        @Override
63        public void handleError(ClientHttpResponse response) throws IOException {
64          log.debug("errorHandler.handleError() called");
65          log.debug(response.getStatusText());
66  
67          String json = response.getStatusText();
68          ObjectMapper mapper = new ObjectMapper();
69          GwApiError gwApiError = mapper.readValue(json, GwApiError.class);
70          throw gwApiError;
71        }
72      };
73  
74      this.setErrorHandler(errorHandler);
75    }
